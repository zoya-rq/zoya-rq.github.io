<html>
  <head>
     <link rel="stylesheet" type="text/css"
           href="mystyle.css">
  </head>

Inline CSS
<!DOCTYPE html>
<html>
   <body style="background-color: palegreen">
        <h1 style="color: brown;">
            This is a heading
        </h1>
        <p style="color: purple">
            This is a paragraph.
        </p>
   </body>
</html>

<html>
   <head>
      <style>
         body {
            background-color: linen;
         }
         h1 {
            color: maroon;
         }
      </style>
   </head>
   <body>
   </body>
</html>
